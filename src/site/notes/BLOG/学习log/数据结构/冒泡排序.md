~~~c
for(end = lenthofarray; end > 0 ;end++)
{
	for(i = 0;i < end; i++)
	{
		if(array[i]>array[i+1])//交换
		{
			a = array[i];
			array[i] = array[i+1];
			array[i+1] = array[i];
		}
	}
}
~~~
